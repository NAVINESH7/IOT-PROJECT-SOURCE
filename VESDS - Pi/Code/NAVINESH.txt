#!/usr/bin/env python3
import cv2
import smbus2 as smbus
import time
import math
import RPi.GPIO as GPIO
from scipy.spatial import distance
from RPLCD.i2c import CharLCD
import serial
import sys

# ===============================
# ----------- SMS CONFIG --------
# ===============================
UART_PORT = '/dev/serial0'
BAUD = 115200
PHONE_NUMBER = '+918122735200'
LOCATION = "11.3628, 77.8279"
MAP_LINK = "https://maps.app.goo.gl/jANYPc3NRyoTDEg66"

def open_serial(port, baud, timeout=1):
    try:
        ser = serial.Serial(port, baud, timeout=timeout)
        ser.reset_input_buffer()
        ser.reset_output_buffer()
        return ser
    except:
        return None

def at_cmd(ser, cmd, wait=1.0):
    if ser is None: return None
    ser.write((cmd + '\r\n').encode())
    time.sleep(wait)
    return ser.read_all().decode(errors='ignore')

def send_sms(ser, phone, text):
    if ser is None: return False
    at_cmd(ser, 'AT+CMGF=1')
    at_cmd(ser, f'AT+CMGS="{phone}"')
    ser.write((text + chr(26)).encode())
    time.sleep(3)
    return True

# ===============================
# -------- GPIO SETUP -----------
# ===============================
GPIO.setmode(GPIO.BCM)
RELAY_PIN = 17
MQ3_PIN = 27
GPIO.setup(RELAY_PIN, GPIO.OUT)
GPIO.setup(MQ3_PIN, GPIO.IN)

BUZZER_PIN = 21
GPIO.setup(BUZZER_PIN, GPIO.OUT)

GPIO_TRIG = 11
GPIO_ECHO = 24
GPIO.setup(GPIO_TRIG, GPIO.OUT)
GPIO.setup(GPIO_ECHO, GPIO.IN)
GPIO.output(GPIO_TRIG, GPIO.LOW)
time.sleep(2)

lcd = CharLCD('PCF8574', 0x27)
lcd.clear()
lcd.write_string("Initializing...")

# MPU6050
MPU6050_ADDR = 0x68
PWR_MGMT_1 = 0x6B
bus = smbus.SMBus(1)
bus.write_byte_data(MPU6050_ADDR, PWR_MGMT_1, 0)

def read_raw_data(addr):
    high = bus.read_byte_data(MPU6050_ADDR, addr)
    low = bus.read_byte_data(MPU6050_ADDR, addr+1)
    val = (high << 8) | low
    if val > 32768: val -= 65536
    return val

def measure_distance():
    GPIO.output(GPIO_TRIG, True)
    time.sleep(0.00001)
    GPIO.output(GPIO_TRIG, False)
    start_time = time.time()
    while GPIO.input(GPIO_ECHO) == 0:
        start_time = time.time()
    while GPIO.input(GPIO_ECHO) == 1:
        bounce_time = time.time()
    pulse = bounce_time - start_time
    return round(pulse * 17150, 2)

# ===============================
# -------- CAMERA SETUP ---------
# ===============================
cap = cv2.VideoCapture(0)
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades+"haarcascade_frontalface_default.xml")
eye_cascade = cv2.CascadeClassifier(cv2.data.haarcascades+"haarcascade_eye.xml")

def Detect_Eye(eye):
    (ex,ey,ew,eh) = eye
    top = (ex+ew//2, ey)
    bottom = (ex+ew//2, ey+eh)
    left = (ex, ey+eh//2)
    right = (ex+ew, ey+eh//2)
    return distance.euclidean(top,bottom) / max(distance.euclidean(left,right),1)

def countdown_alert():
    for i in range(10, -1, -1):
        lcd.clear()
        lcd.write_string("ALERT!\nCountdown: " + str(i))
        print("Alert countdown:", i)
        GPIO.output(BUZZER_PIN, GPIO.LOW)
        time.sleep(1)
    GPIO.output(BUZZER_PIN, GPIO.HIGH)

# ===============================
# ---------- MAIN LOOP ---------
# ===============================
lcd.clear()
lcd.write_string("System Active")
ser = open_serial(UART_PORT, BAUD)

while True:
    alerts = []
    
    # Alcohol
    if GPIO.input(MQ3_PIN) == GPIO.HIGH:
        alerts.append("Alcohol Detected!")
    
    # Accident
    ax = read_raw_data(0x3B)/16384.0
    ay = read_raw_data(0x3D)/16384.0
    az = read_raw_data(0x3F)/16384.0
    total_acc = math.sqrt(ax*ax + ay*ay + az*az)
    if total_acc > 2.0:
        alerts.append("Accident!")
    
    # Ultrasonic (50cm)
    dist = measure_distance()
    if dist < 10:
        ultrasonic_alert = True
    else:
        ultrasonic_alert = False

    # Drowsy Detection (more sensitive)
    ret, frame = cap.read()
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    faces = face_cascade.detectMultiScale(gray, 1.2, 4)
    status = "Normal"
    for (x,y,w,h) in faces:
        roi = gray[y:y+h, x:x+w]
        eyes = eye_cascade.detectMultiScale(roi, 1.2, 4)
        if len(eyes) >= 2:
            ear = sum(Detect_Eye(e) for e in eyes) / len(eyes)
            if ear < 0.28: status="Drowsy"
        else:
            status="Drowsy"
    if status == "Drowsy":
        alerts.append("Drowsy!")
    
    # Handle Alerts
    if alerts:
        lcd.clear()
        lcd.write_string(alerts[0])
        print(alerts)

        msg = "\n".join(alerts) + f"\nLocation: {LOCATION}\n{MAP_LINK}"
        send_sms(ser, PHONE_NUMBER, msg)

        countdown_alert()
    
    else:
        # Normal OR ultrasonic
        if ultrasonic_alert:
            lcd.clear()
            lcd.write_string(f"Obstacle {dist}cm")
            GPIO.output(BUZZER_PIN, GPIO.LOW)
        else:
            GPIO.output(BUZZER_PIN, GPIO.HIGH)
            lcd.clear()
            lcd.write_string(f"Safe {dist}cm")

    cv2.imshow("Driver Monitor", frame)
    if cv2.waitKey(10) & 0xFF == 27:
        break

GPIO.cleanup()
cap.release()
cv2.destroyAllWindows()
lcd.clear()
lcd.write_string("Stopped")
