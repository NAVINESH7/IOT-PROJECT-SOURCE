#!/usr/bin/env python3
import cv2
import smbus2 as smbus
import time
import math
import RPi.GPIO as GPIO
from scipy.spatial import distance
from RPLCD.i2c import CharLCD
import serial
import sys
import requests

# ===============================
# ----------- FIREBASE CONFIG ---
# ===============================
FIREBASE_URL = "https://vision-enhance-safety-platform-default-rtdb.firebaseio.com/vehicle_data.json"

def send_to_firebase(data):
    try:
        r = requests.post(FIREBASE_URL, json=data)
        if r.status_code == 200:
            print("[FB] Data uploaded:", r.json())
        else:
            print("[FB] Error:", r.status_code, r.text)
    except Exception as e:
        print("[FB] Exception:", e)

# ===============================
# ----------- GSM / GPS CONFIG --
# ===============================
UART_PORT = '/dev/serial0'
BAUD = 9600          # Many SIM808 modules default to 9600. Change if needed.
PHONE_NUMBER = '+918122735200'

def open_serial(port, baud, timeout=1):
    try:
        ser = serial.Serial(port, baud, timeout=timeout)
        ser.reset_input_buffer()
        ser.reset_output_buffer()
        print("[SERIAL] Opened", port)
        return ser
    except Exception as e:
        print("[SERIAL] Open failed:", e)
        return None

def at_cmd(ser, cmd, wait=1.0):
    if ser is None:
        return ""
    ser.write((cmd + '\r\n').encode())
    time.sleep(wait)
    try:
        resp = ser.read_all().decode(errors='ignore')
    except:
        resp = ""
    # print(f"[AT] {cmd} -> {resp.strip()}")
    return resp

def init_gps(ser):
    # Power on GNSS (SIM808 GNSS engine)
    print("[GPS] Powering GPS ON...")
    at_cmd(ser, "AT")              # wake
    at_cmd(ser, "AT+CGNSPWR=1", 1) # power GNSS
    at_cmd(ser, "AT+CGNSSEQ=\"RMC\"", 1)  # optional, GNSS NMEA sentence config
    print("[GPS] Init commands sent. Waiting for fix...")

def get_gps_location(ser):
    """
    Uses AT+CGNSINF to get GPS info.
    Typical response:
    +CGNSINF: 1,1,20250101120000.000,11.362800,77.827900,123.4,0.00,0.0,1,,1.0,1.0,1.0,,,
    Fields: <run_status>,<fix_status>,<UTC>,<lat>,<lon>,...
    We use fix_status, lat, lon.
    """
    resp = at_cmd(ser, "AT+CGNSINF", 1.0)
    lat = lon = None

    for line in resp.splitlines():
        line = line.strip()
        if line.startswith("+CGNSINF:"):
            try:
                parts = line.split(":")[1].strip().split(",")
                run_status = parts[0]           # '1' if GNSS running
                fix_status = parts[1]           # '1' if fix acquired
                if run_status == '1' and fix_status == '1':
                    lat = parts[3]
                    lon = parts[4]
            except Exception as e:
                print("[GPS] Parse error:", e)
            break

    if lat and lon and lat != "" and lon != "":
        print(f"[GPS] Fix: lat={lat}, lon={lon}")
        return lat, lon
    else:
        print("[GPS] No valid fix yet")
        return None, None

def send_sms(ser, phone, text):
    if ser is None:
        print("[SMS] Serial not available")
        return False
    try:
        at_cmd(ser, "AT")
        at_cmd(ser, "AT+CMGF=1")          # text mode
        at_cmd(ser, f'AT+CMGS="{phone}"')
        ser.write((text + chr(26)).encode())  # Ctrl+Z
        time.sleep(3)
        print("[SMS] Sent")
        return True
    except Exception as e:
        print("[SMS] Error:", e)
        return False

# ===============================
# -------- GPIO SETUP -----------
# ===============================
GPIO.setmode(GPIO.BCM)
RELAY_PIN = 17
MQ3_PIN = 27
BUZZER_PIN = 21
GPIO_TRIG = 11
GPIO_ECHO = 24

GPIO.setup(RELAY_PIN, GPIO.OUT)
GPIO.setup(MQ3_PIN, GPIO.IN)
GPIO.setup(BUZZER_PIN, GPIO.OUT)
GPIO.setup(GPIO_TRIG, GPIO.OUT)
GPIO.setup(GPIO_ECHO, GPIO.IN)
GPIO.output(GPIO_TRIG, GPIO.LOW)
time.sleep(2)

lcd = CharLCD('PCF8574', 0x27)
lcd.clear()
lcd.write_string("Initializing...")

# ===============================
# -------- MPU6050 SETUP --------
# ===============================
MPU6050_ADDR = 0x68
PWR_MGMT_1 = 0x6B
bus = smbus.SMBus(1)
bus.write_byte_data(MPU6050_ADDR, PWR_MGMT_1, 0)

def read_raw_data(addr):
    high = bus.read_byte_data(MPU6050_ADDR, addr)
    low = bus.read_byte_data(MPU6050_ADDR, addr + 1)
    value = (high << 8) | low
    if value > 32768:
        value -= 65536
    return value

def measure_distance():
    GPIO.output(GPIO_TRIG, True)
    time.sleep(0.00001)
    GPIO.output(GPIO_TRIG, False)
    start_time = time.time()
    while GPIO.input(GPIO_ECHO) == 0:
        start_time = time.time()
    while GPIO.input(GPIO_ECHO) == 1:
        bounce_back = time.time()
    pulse_duration = bounce_back - start_time
    distance = pulse_duration * 17150
    return round(distance, 2)

# ===============================
# -------- CAMERA SETUP ---------
# ===============================
cap = cv2.VideoCapture(0)
if not cap.isOpened():
    print("Cannot open camera. Check connection.")
    sys.exit(1)

face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades +
                                     "haarcascade_frontalface_default.xml")
eye_cascade = cv2.CascadeClassifier(cv2.data.haarcascades +
                                    "haarcascade_eye.xml")

def Detect_Eye(eye):
    (ex, ey, ew, eh) = eye
    top = (ex + ew//2, ey)
    bottom = (ex + ew//2, ey + eh)
    left = (ex, ey + eh//2)
    right = (ex + ew, ey + eh//2)
    return distance.euclidean(top, bottom) / max(distance.euclidean(left, right), 1)

def countdown_alert():
    # 10-second countdown with buzzer beeps and LCD update
    for i in range(10, -1, -1):
        lcd.clear()
        lcd.write_string(f"ALERT!\nCountdown: {i}")
        GPIO.output(BUZZER_PIN, GPIO.LOW)   # buzzer ON
        time.sleep(0.5)
        GPIO.output(BUZZER_PIN, GPIO.HIGH)  # buzzer OFF
        time.sleep(0.5)

# ===============================
# ---------- MAIN LOOP ----------
# ===============================
lcd.clear()
lcd.write_string("System Active")

ser = open_serial(UART_PORT, BAUD)
if ser:
    init_gps(ser)

last_lat = None
last_lon = None

try:
    while True:
        alerts = []
        ultrasonic_alert = False

        # -------- Alcohol Detection --------
        alcohol_detected = (GPIO.input(MQ3_PIN) == GPIO.HIGH)
        if alcohol_detected:
            alerts.append("Alcohol Detected!")

        # -------- Accident Detection --------
        acc_x = read_raw_data(0x3B) / 16384.0
        acc_y = read_raw_data(0x3D) / 16384.0
        acc_z = read_raw_data(0x3F) / 16384.0
        total_acc = math.sqrt(acc_x**2 + acc_y**2 + acc_z**2)
        accident_detected = total_acc > 2.0
        if accident_detected:
            alerts.append("Accident Detected!")

        # -------- Ultrasonic Detection (<10cm) --------
        distance_cm = measure_distance()
        if distance_cm < 10:
            ultrasonic_alert = True

        # -------- Drowsiness Detection --------
        ret, frame = cap.read()
        if not ret:
            print("Camera read error.")
            break

        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        faces = face_cascade.detectMultiScale(gray, 1.1, 5)
        status = "Normal"
        drowsy_detected = False

        for (x, y, w, h) in faces:
            roi = gray[y:y+h, x:x+w]
            eyes = eye_cascade.detectMultiScale(roi, 1.1, 5)
            if len(eyes) >= 2:
                ear_list = [Detect_Eye(e) for e in eyes]
                if sum(ear_list)/len(ear_list) < 0.28:
                    status = "Drowsy"
            else:
                status = "Drowsy"

        if status == "Drowsy":
            drowsy_detected = True
            alerts.append("Driver Drowsy!")

        # -------- GPS Location (try every loop) --------
        lat, lon = get_gps_location(ser) if ser else (None, None)
        if lat and lon:
            last_lat, last_lon = lat, lon

        # Build map link from latest known coordinates
        if last_lat and last_lon:
            map_link = f"https://www.google.com/maps?q={last_lat},{last_lon}"
        else:
            map_link = "GPS FIX NOT AVAILABLE"

        # -------- Prepare Firebase JSON --------
        fb_data = {
            "alcohol": "ALCOHOL DETECTED" if alcohol_detected else "NORMAL",
            "drowsy": "DROWSY DETECTED" if drowsy_detected else "NORMAL",
            "accident": "ACCIDENT DETECTED" if accident_detected else "NORMAL",
            "distance_cm": distance_cm,
            "force_g": round(total_acc, 2),
            "latitude": last_lat if last_lat else "N/A",
            "longitude": last_lon if last_lon else "N/A",
            "map_link": map_link,
            "timestamp": time.strftime("%Y-%m-%d %H:%M:%S")
        }

        # -------- Send to Firebase --------
        send_to_firebase(fb_data)

        # -------- Local Handling (LCD, Relay, SMS) --------
        if alerts:
            lcd.clear()
            lcd.write_string(alerts[0])

            sms_text = (
                "⚠️ Vehicle Safety Alert!\n\n"
                "Event Detected:\n"
                + "\n".join(alerts) +
                "\n\nDriver may be in danger 🚨\n\n"
            )

            if last_lat and last_lon:
                sms_text += f"📍Location:\nLat: {last_lat}, Lon: {last_lon}\n{map_link}\n"
            else:
                sms_text += "📍Location: GPS not fixed yet.\n"

            sms_text += "\nPlease take immediate action!"

            send_sms(ser, PHONE_NUMBER, sms_text)

            countdown_alert()
            GPIO.output(RELAY_PIN, GPIO.HIGH)   # stop engine after countdown

        elif ultrasonic_alert:
            lcd.clear()
            lcd.write_string(f"Obstacle {distance_cm}cm")
            GPIO.output(BUZZER_PIN, GPIO.LOW)   # buzzer ON
            GPIO.output(RELAY_PIN, GPIO.LOW)    # engine ON

        else:
            lcd.clear()
            lcd.write_string(f"Safe {distance_cm}cm")
            GPIO.output(RELAY_PIN, GPIO.LOW)    # engine ON
            GPIO.output(BUZZER_PIN, GPIO.HIGH)  # buzzer OFF

        cv2.imshow("Driver Monitor", frame)
        if cv2.waitKey(10) & 0xFF == 27:
            break

        time.sleep(0.5)

except KeyboardInterrupt:
    print("Program stopped by user")

finally:
    GPIO.output(RELAY_PIN, GPIO.LOW)
    GPIO.output(BUZZER_PIN, GPIO.HIGH)
    GPIO.cleanup()
    cap.release()
    cv2.destroyAllWindows()
    lcd.clear()
    lcd.write_string("System Stopped")
    if ser:
        ser.close()
